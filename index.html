var has function scope

                            
function printNumbers () {
    for(var i = 0; i < 5; i++) {
        setTimeout(function(){
            console.log(i);
        }, i * 50);
    }

    console.log(i); // 5
}

let has block scope


function printNumbers () {
    for(let i = 0; i < 5; i++) {
        setTimeout(function(){
            console.log(i);
        }, i * 50);
    }

    console.log(i); // Uncaught ReferenceError: i is not defined
}
